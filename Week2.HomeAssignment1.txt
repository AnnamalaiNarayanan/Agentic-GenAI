import { test, expect } from '@playwright/test';

test('Create Lead in Leaftaps', async ({ page }) => {
  // ---------- Step 1: Launch Application ----------
  await page.goto('http://leaftaps.com/opentaps/control/main');
  await page.setViewportSize({ width: 1600, height: 900 });

  // ---------- Step 2: Login ----------
  const usernameField = page.locator('#username');
  const passwordField = page.locator('input[name="PASSWORD"]');
  const loginButton = page.locator('.decorativeSubmit');

  await usernameField.fill('Demosalesmanager');
  await passwordField.fill('crmsfa');
  await loginButton.click();

  // ---------- Step 3: Navigate to CRM/SFA ----------
  await page.locator('text=CRM/SFA').click();

  // ---------- Step 4: Navigate to Leads → Create Lead ----------
  await page.locator('text=Leads').click();
  await page.locator('text=Create Lead').click();

  // ---------- Step 5: Fill Lead Form ----------
  const companyNameField = page.locator('#createLeadForm_companyName');
  const firstNameField = page.locator('#createLeadForm_firstName');
  const lastNameField = page.locator('#createLeadForm_lastName');
  const sourceDropdown = page.locator('#createLeadForm_dataSourceId');
  const industryDropdown = page.locator('#createLeadForm_industryEnumId');
  const ownershipDropdown = page.locator('#createLeadForm_ownershipEnumId');

  await companyNameField.fill('TestLeaf');
  await firstNameField.fill('Gokul');
  await lastNameField.fill('Sekar');

  await sourceDropdown.selectOption({ index: 4 });
  await industryDropdown.selectOption({ label: 'Computer Software' });
  await ownershipDropdown.selectOption({ value: 'OWN_PARTNERSHIP' });

  // ---------- Step 6: Submit Lead ----------
  await page.locator('.smallSubmit').click();

  // ---------- Step 7: Capture Lead ID ----------
  const leadId = await page.locator('#viewLead_companyName_sp').innerText();
  console.log('Lead Id:', leadId);

  // ---------- Step 8: Validate Page Title ----------
  const pageTitle = await page.title();
  console.log('Page Title:', pageTitle);
  await expect(page).toHaveTitle(/View Lead/);

  // ---------- Step 9: Close Browser ----------
  await page.close();
});
